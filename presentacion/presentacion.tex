\section{Introducción}

\begin{frame}{¿Qué es el proyecto?}
\begin{center}
% IMAGEN: Captura de pantalla completa de la aplicación funcionando
% Mostrar la interfaz con el mapa, sidebar y controles visibles
\includegraphics[width=0.85\textwidth]{01-aplicacion-completa.png}
\end{center}
\vspace{0.3cm}
\begin{itemize}
\item Aplicación web interactiva para visualización geoespacial
\item Análisis de variables meteorológicas y de vegetación
\item Generación de animaciones temporales
\end{itemize}
\end{frame}

\begin{frame}{Objetivo Principal}
\begin{center}
% IMAGEN: Diagrama conceptual simple mostrando: Usuario -> Aplicación Web -> Google Earth Engine -> Visualización
% Puede ser un diagrama de flujo simple con flechas
\includegraphics[width=0.7\textwidth]{02-diagrama-flujo.png}
\end{center}
\vspace{0.3cm}
\begin{block}{Objetivo}
Desarrollar una plataforma web que permita visualizar y analizar datos meteorológicos y de vegetación mediante la integración de tecnologías de procesamiento geoespacial en la nube.
\end{block}
\end{frame}
